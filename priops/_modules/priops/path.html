    """A sequence of edges, together with the net weight."""


[docs]    def __init__(self, edges):
        """*edges* is a list of Edges."""

        list.__init__(self, edges)

[docs]    def __str__(self):
        """Returns pretty-formatted string containing a numbered list of 
        edges making up the Path."""
        result = '(Path:'
        for (i, element) in enumerate(self):
            result += '\n%d. %s' % (i + 1, element)
        result += '\n)'
        return result

        """Combines this path with *edge* to a new path.  *edge* will extend 
        *self* at the end of *self*."""

        return Path(self + [edge])



[docs]    def __rxor__(self, edge):
        """Combines this path with *edge* to a new path.  *edge* will extend
        *self* at the start of *self*."""

        return Path([edge] + self)



[docs]    def __call__(self, input):
        """Processes inter-node data *input*."""

        result = input
        for edge in self:
