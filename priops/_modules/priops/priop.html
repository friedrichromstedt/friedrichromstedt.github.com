[docs]    def __add__(self, other_priop):
        """Concatenates both the list items as well as the names.  If the 
        names are the same, the name is kept."""

        if self.name == other_priop.name:
            return Priop(list.__add__(self, other_priop), name=self.name)
        else:
            return Priop(list.__add__(self, other_priop),
                name=', '.join([self.name, other_priop.name]))
